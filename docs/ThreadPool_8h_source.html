319  {
320  std::lock_guard<std::mutex> lock(mutex_);
321 
322  if (is_full_i())

332  PackagedTask{ std::move(fct) }, std::move(name));
333 
334  TaskHandle<Result> handle{
335  next_task_id_, named_task_ptr->fct_.get_future(), shared_from_this() };
336
